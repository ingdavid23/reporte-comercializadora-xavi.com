// ================= ENVÍO DE REPORTE (EMPLEADO) ================= window.submitReport = async function() { // 1. Destrabar el botón a la fuerza por si se quedó congelado antes isSubmitting = false; if (!currentUser) { alert("Error: Tu sesión ha caducado. Por favor, cierra sesión y vuelve a entrar."); return; } const branch = getBranches().find(b => b.branch_name === currentUser.branch_name); if (!branch) { alert("Error: No se encontró la sucursal asignada a tu usuario."); return; } // 2. Extraer datos con seguridad para evitar valores nulos (NaN) const getVal = (id) => { const el = document.getElementById(id); return el && el.value ? parseInt(el.value) : 0; }; const consumedBultos = getVal('maseca-bultos') + getVal('harimasa-blanca-bultos') + getVal('harimasa-amarilla-bultos') + getVal('harimasa-fardo-bultos'); const consumedKilos = getVal('maseca-kilos') + getVal('harimasa-blanca-kilos') + getVal('harimasa-amarilla-kilos') + getVal('harimasa-fardo-kilos'); if (consumedBultos === 0 && consumedKilos === 0) { alert("⚠️ Debes capturar al menos un consumo de harina para poder enviar el reporte."); return; } // 3. Bloquear el botón visualmente para evitar dobles envíos isSubmitting = true; const btn = document.getElementById('submit-report-btn'); if (btn) { btn.innerHTML = ' Enviando...'; btn.disabled = true; } try { const today = new Date(); const dateStr = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}-${String(today.getDate()).padStart(2, '0')}`; const totalConsumedKilos = (consumedBultos * 20) + consumedKilos; const totalInitialKilos = ((branch.initial_stock_bultos || 0) * 20) + (branch.initial_stock_kilos || 0); let newTotalKilos = Math.max(0, totalInitialKilos - totalConsumedKilos); const newBultos = Math.floor(newTotalKilos / 20); const newKilos = newTotalKilos % 20; const isChecked = (name) => { const radio = document.querySelector(`input[name="${name}"]:checked`); return radio ? radio.value === 'si' : false; }; const reportData = { type: 'report', report_date: dateStr, branch_id: branch.__backendId || currentUser.branch_name, branch_name: currentUser.branch_name, initial_stock_bultos: branch.initial_stock_bultos || 0, initial_stock_kilos: branch.initial_stock_kilos || 0, maseca_premium_bultos: getVal('maseca-bultos'), maseca_premium_kilos: getVal('maseca-kilos'), harimasa_blanca_bultos: getVal('harimasa-blanca-bultos'), harimasa_blanca_kilos: getVal('harimasa-blanca-kilos'), harimasa_amarilla_bultos: getVal('harimasa-amarilla-bultos'), harimasa_amarilla_kilos: getVal('harimasa-amarilla-kilos'), harimasa_fardo_bultos: getVal('harimasa-fardo-bultos'), harimasa_fardo_kilos: getVal('harimasa-fardo-kilos'), tiene_papel: isChecked('papel'), tiene_teflon: isChecked('teflon'), tiene_grafito: isChecked('grafito'), existencia_final_bultos: newBultos, existencia_final_kilos: newKilos, submitted_by: currentUser.username, submitted_at: new Date().toISOString() }; // 4. Subida a la nube await addDoc(collection(db, 'reports'), reportData); await updateDoc(doc(db, 'branches', branch.__backendId), { initial_stock_bultos: newBultos, initial_stock_kilos: newKilos }); // Usar la alerta nativa si el toast llegara a fallar if (typeof showToast === "function") showToast('¡Reporte enviado exitosamente!'); else alert('¡Reporte enviado exitosamente!'); // 5. 🔥 LIMPIEZA TOTAL POST-ENVÍO 🔥 const inputs = [ 'maseca-bultos', 'maseca-kilos', 'harimasa-blanca-bultos', 'harimasa-blanca-kilos', 'harimasa-amarilla-bultos', 'harimasa-amarilla-kilos', 'harimasa-fardo-bultos', 'harimasa-fardo-kilos' ]; inputs.forEach(id => { const el = document.getElementById(id); if (el) el.value = ''; }); ['papel', 'teflon', 'grafito'].forEach(name => { const noRadio = document.querySelector(`input[name="${name}"][value="no"]`); if (noRadio) noRadio.checked = true; }); if (typeof calculateFinalStock === "function") calculateFinalStock(); const scroll = document.getElementById('employee-scroll'); if (scroll) scroll.scrollTo({ top: 0, behavior: 'smooth' }); } catch (error) { console.error("Fallo al enviar:", error); alert("❌ Ocurrió un error al guardar el reporte: " + error.message); } finally { // Liberar el botón siempre, sin importar si hubo éxito o error isSubmitting = false; if (btn) { btn.innerHTML = ' Enviar Reporte Oficial'; btn.disabled = false; if (typeof lucide !== 'undefined') lucide.createIcons(); } } };
